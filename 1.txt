import { ComponentFixture, TestBed } from '@angular/core/testing';
import { FormsModule } from '@angular/forms';
import { of } from 'rxjs';
import { ViewComponent } from './view.component';
import { AppService } from '../app.service';

describe('ViewComponent', () => {
  let component: ViewComponent;
  let fixture: ComponentFixture<ViewComponent>;
  let appServiceMock: any;

  beforeEach(async () => {
    appServiceMock = {
      getGitHubUser: jasmine.createSpy('getGitHubUser').and.returnValue(
        of({
          login: 'johnDoe',
          id: 123,
          public_repos: 10,
          avatar_url: 'http://example.com/avatar.jpg',
        })
      ),
    };

    await TestBed.configureTestingModule({
      declarations: [ViewComponent],
      imports: [FormsModule],
      providers: [{ provide: AppService, useValue: appServiceMock }],
    }).compileComponents();
  });

  beforeEach(() => {
    fixture = TestBed.createComponent(ViewComponent);
    component = fixture.componentInstance;
    fixture.detectChanges();
  });

  it('should create the component', () => {
    expect(component).toBeTruthy();
  });

  it('should call service and fetch user data on search', () => {
    component.userName = 'johnDoe';
    component.search();

    expect(appServiceMock.getGitHubUser).toHaveBeenCalledWith('johnDoe');
    expect(component.response).toEqual({
      login: 'johnDoe',
      id: 123,
      public_repos: 10,
      avatar_url: 'http://example.com/avatar.jpg',
    });
  });
});
