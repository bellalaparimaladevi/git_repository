import { Injectable } from '@angular/core';
import { BehaviorSubject, Observable } from 'rxjs';
import { HttpClient } from '@angular/common/http';

@Injectable({
  providedIn: 'root',
})
export class GithubService {
  private searchTermSubject = new BehaviorSubject<string>(''); // Initialize with an empty string
  searchTerm$ = this.searchTermSubject.asObservable(); // Expose as Observable

  constructor(private http: HttpClient) {}

  // Update the search term
  setSearchTerm(term: string) {
    this.searchTermSubject.next(term); // Emit the new term
  }

  // Fetch user data from GitHub API
  getUser(username: string): Observable<any> {
    return this.http.get(`https://api.github.com/users/${username}`);
  }
}
